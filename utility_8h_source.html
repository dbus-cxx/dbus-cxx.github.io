141 template<typename arg1, typename... argn>
142 class method_signature<arg1, argn...> : public method_signature<argn...> {
143 public:
144  std::string method_sig() const {
145  std::string arg1_name = demangle<arg1>();
146  std::string remaining_args = method_signature<argn...>::method_sig();
147 
148  if( remaining_args.size() > 1 ) {
149  arg1_name += ",";
150  }
151 
152  return arg1_name + remaining_args;
153  }

155  std::string introspect( const std::vector<std::string>& names, int idx, const std::string& spaces ) const {
156  arg1 arg;
157  std::ostringstream output;
158  std::string name = names.size() > idx ? names[idx] : "";
159  output << spaces;
160  output << "<arg ";
161 
162  if( name.size() > 0 ) { output << "name=\"" << name << "\" "; }
163 
164  output << "type=\"" << signature( arg ) << "\" ";
165  output << "direction=\"in\"/>\n";
166  output << method_signature<argn...>().introspect( names, idx + 1, spaces );
167  return output.str();
168  }
