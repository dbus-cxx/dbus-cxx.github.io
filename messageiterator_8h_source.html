207  template <typename T>
208  void get_array( std::vector<T>& array ) {
209  if( !this->is_array() ) { /* Should never happen */
210  throw ErrorInvalidTypecast( "MessageIterator: Extracting non array into std::vector" );
211  }
212 
213  array.clear();
214 
215  MessageIterator subiter = this->recurse();
216 
217  while( subiter.is_valid() ) {
218  //NOTE: we don't need to do subiter.next() here, because
219  //operator>> does that for us
220  T val;
221  subiter >> val;
222  array.push_back( val );
223  }
224  }

235  template <typename Key, typename Data>
236  void get_dict( std::map<Key, Data>& dict ) {
237  Key val_key;
238  Data val_data;
239 
240  MessageIterator subiter = this->recurse();
241 
242  while( subiter.is_valid() ) {
243  MessageIterator subSubiter = subiter.recurse();
244 
245  // When we recurse the second time, our demarshaling gets aligned.
246  // This may cause the subiter to then become invalid because of
247  // the new byte offset, so we need a second is_valid() check here.
248  if( !subiter.is_valid() ){
249  break;
250  }
251 
252  while( subSubiter.is_valid() ) {
253  subSubiter >> val_key;
254  subSubiter >> val_data;
255  dict[ val_key ] = val_data;
256  subSubiter.next();
257  }
258 
259  subiter.next();
260  }
261  }
