204  template <typename T>
205  void get_array( std::vector<T>& array ) {
206  if( !this->is_array() ) { /* Should never happen */
207  throw ErrorInvalidTypecast( "MessageIterator: Extracting non array into std::vector" );
208  }
209 
210  array.clear();
211 
212  MessageIterator subiter = this->recurse();
213 
214  while( subiter.is_valid() ) {
215  //NOTE: we don't need to do subiter.next() here, because
216  //operator>> does that for us
217  T val;
218  subiter >> val;
219  array.push_back( val );
220  }
221  }

224  void get_struct( std::tuple<T...>& tup ) {
225  MessageIterator subiter = this->recurse();
226  std::apply( [subiter]( auto&& ...arg ) mutable {
227  ( subiter >> ... >> arg );

232  template <typename Key, typename Data>
233  void get_dict( std::map<Key, Data>& dict ) {
234  Key val_key;
235  Data val_data;
236 
237  MessageIterator subiter = this->recurse();
238 
239  while( subiter.is_valid() ) {
240  MessageIterator subSubiter = subiter.recurse();
241 
242  while( subSubiter.is_valid() ) {
243  subSubiter >> val_key;
244  subSubiter >> val_data;
245  dict[ val_key ] = val_data;
246  subSubiter.next();
247  }
248 
249  subiter.next();
250  }
251  }
