176 template <typename Key, typename Data> inline std::string signature( const std::map<Key, Data>& ) {
177  Key k; Data d;
178  std::string sig;
179  sig = DBUSCXX_TYPE_ARRAY_AS_STRING;
180  sig += DBUSCXX_DICT_ENTRY_BEGIN_CHAR_AS_STRING +
181  signature( k ) + signature( d ) +
182  DBUSCXX_DICT_ENTRY_END_CHAR_AS_STRING;
183  return sig;
184 }
185 
186 //Note: we need to have two different signature() methods for dictionaries; this is because
187 //when introspecting, we need to use the normal signature() so that it comes up properly.
188 //However, when we are sending out data, that signature would give us an extra array signature,
189 //which is not good. Hence, this method is only used when we need to send out a dict
190 template <typename Key, typename Data> inline std::string signature_dict_data( const std::map<Key, Data>& ) {
191  Key k; Data d;
192  std::string sig;
193  sig = DBUSCXX_DICT_ENTRY_BEGIN_CHAR_AS_STRING +
194  signature( k ) + signature( d ) +
195  DBUSCXX_DICT_ENTRY_END_CHAR_AS_STRING;
196  return sig;
197 }
